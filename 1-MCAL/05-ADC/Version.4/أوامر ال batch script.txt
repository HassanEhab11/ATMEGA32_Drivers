ver		: نوع النظام واصداره  
cd		: الانتقال الي الملف
cd..	: الرجوع الي الخلف
cls		: مسح كل اللي في ال cmd 
del {file name}	: مسح الملف
dir {link}	: قراءة الملفات
echo {xxxxx}	: طباعة المراد طباعتة
exit	: الخروج من الدوس 
md {file name}	: عمل ملف باسم 
move {from} {to}	: نقل الملف من والي
pause	: ايقاف البرنامج موقتا لحين الضغط علي اي زر 
rd {file name}	: مسح الملف بس لازم يكون فاضي 
ren {file name} {new file}	: اعادة تسمية الملفات
start {filename.xx}	: تشغيل البرنامج
%time%	:	اظهار الوقت
%date%	:	اظهار التاريخ
copy {from} {to}	: نسخ ملف من الي
